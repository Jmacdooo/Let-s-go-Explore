function drawDog() { const sx = dog.x - camera.x; const sy = dog.y - camera.y - 10; // lift dog slightly off ground let img; if (dog.state === "walk") img = images["dog_walk"]; else if (dog.state === "eat") img = images["dog_eat"]; else img = images["dog_idle"]; if (!img) return; // ---- WALKING (sprite strip) ---- if (dog.state === "walk") { const FRAMES = 6; // your walk strip const frameW = img.width / FRAMES; const frameH = img.height; const frame = Math.floor(dog.animTime / 6) % FRAMES; const drawH = 210; const drawW = drawH * (frameW / frameH); // âœ… IMPORTANT: flip ONLY when moving LEFT const flip = (dog.dir === -1); ctx.save(); if (flip) { ctx.translate(sx + drawW, sy); ctx.scale(-1, 1); ctx.drawImage( img, frame * frameW, 0, frameW, frameH, 0, 0, drawW, drawH ); } else { ctx.drawImage( img, frame * frameW, 0, frameW, frameH, sx, sy, drawW, drawH ); } ctx.restore(); return; } // ---- IDLE / EAT (single PNG) ---- const frameW = img.width; const frameH = img.height; const drawH = 210; const drawW = drawH * (frameW / frameH); const flip = (dog.dir === -1); // flip ONLY on left ctx.save(); if (flip) { ctx.translate(sx + drawW, sy); ctx.scale(-1, 1); ctx.drawImage(img, 0, 0, frameW, frameH, 0, 0, drawW, drawH); } else { ctx.drawImage(img, sx, sy, drawW, drawH); } ctx.restore(); }
